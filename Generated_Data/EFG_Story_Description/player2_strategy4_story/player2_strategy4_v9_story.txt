Adjusted Payoff Calculation:
def adjusted_payoffs(i, j):
    raw = [base_payoffs['Player1'][i], base_payoffs['Player2'][j]]
    avg = sum(raw) / 2
    return [round(p - avg, 2) for p in raw]
